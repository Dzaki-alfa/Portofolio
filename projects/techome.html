TecHome Smart Home System

Kategori: IoT / Smart Home
Tahun: 2025
Durasi: 3 bulan
Tingkat: Menengah

Komponen Utama

ESP32 DevKit V1 Sensor DHT11 Sensor Gas MQ2 Relay 10 Channel Buzzer Alarm Power Supply 24V

Spesifikasi Sistem

  • Controller ESP32 DevKit V1
  • Koneksi WiFi IoT
  • Sensor DHT11 & MQ2
  • Control Web Dashboard & Mobile

Deskripsi Proyek

TecHome merupakan sistem Smart Home berbasis Internet of Things (IoT) yang dirancang untuk mengontrol serta memantau berbagai perangkat rumah secara otomatis dan real-time. Sistem ini menggunakan ESP32 sebagai mikrokontroler utama yang terhubung ke jaringan internet sehingga perangkat dapat dikontrol dari jarak jauh melalui web dashboard maupun aplikasi mobile.

TecHome mengintegrasikan sensor suhu, kelembapan, serta sensor deteksi gas untuk meningkatkan kenyamanan dan keamanan rumah. Data sensor dikirim ke database MySQL melalui komunikasi HTTP, kemudian ditampilkan pada dashboard monitoring secara real-time.

Tujuan Sistem

Tujuan utama dari pengembangan TecHome adalah menghadirkan sistem rumah pintar yang mampu meningkatkan efisiensi energi, keamanan, dan kenyamanan penghuni rumah. Dengan integrasi sensor dan otomatisasi perangkat, pengguna dapat memantau kondisi rumah secara langsung serta mengendalikan perangkat elektronik dari mana saja.

Fitur Utama

Smart Home Automation

Mengontrol perangkat rumah melalui web atau aplikasi.

Monitoring Suhu

Sensor DHT11 membaca suhu dan kelembapan secara real-time.

Deteksi Gas Berbahaya

Sensor MQ2 mendeteksi gas seperti LPG dan asap.

Sistem Alarm

Buzzer aktif otomatis saat terdeteksi gas berbahaya.

Contoh Program ESP32 



#include < WiFi.h>
#include < WiFiClientSecure.h>
#include < HTTPClient.h>
#include < ArduinoJson.h>
#include < PZEM004Tv30.h>
#include < DHT.h>

const char* ssid = "TEKNOLAB Office";
const char* password = "selamatdatang";
const char* serverName = "https://techome.web.id/simpan.php";
const char* getURL = "https://techome.web.id/get_relay_status.php";

PZEM004Tv30 pzem(Serial2, 16, 17); // RX2, TX2

#define BUZZER_PIN 25
#define DHTPIN 19
#define DHTTYPE DHT11
DHT dht(DHTPIN, DHTTYPE);

#define MQ2_PIN 36
int gasValue = 0;
int relayPins[10] = {23, 22, 32, 33, 5, 26, 27, 14, 21, 13};

WiFiClientSecure client;

void setup() {
  Serial.begin(115200);

  for (int i = 0; i < 10; i++) {
    pinMode(relayPins[i], OUTPUT);
  }

  pinMode(BUZZER_PIN, OUTPUT);
  digitalWrite(BUZZER_PIN, LOW);

  WiFi.begin(ssid, password);
  Serial.print("Menghubungkan ke WiFi");
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\nTerhubung ke WiFi!");

  // sertifikat SSL
  client.setInsecure();

  dht.begin();
  Serial2.begin(9600, SERIAL_8N1, 16, 17);

  Serial.println("Monitoring Sensor Dimulai...");
}

void loop() {
  if (WiFi.status() == WL_CONNECTED) {

    float voltage = pzem.voltage();
    float current = pzem.current();
    float power   = pzem.power();
    float energy  = pzem.energy();

    float suhu = dht.readTemperature();
    float kelembaban = dht.readHumidity();
    gasValue = analogRead(MQ2_PIN);

    if (isnan(voltage))    voltage = 0;
    if (isnan(current))    current = 0;
    if (isnan(power))      power   = 0;
    if (isnan(energy))     energy  = 0;
    if (isnan(suhu))       suhu    = 0;
    if (isnan(kelembaban)) kelembaban = 0;
    if (gasValue < 0)      gasValue = 0;

    Serial.println("Data Sekarang:");
    Serial.print("Tegangan   : "); Serial.println(voltage);
    Serial.print("Arus       : "); Serial.println(current);
    Serial.print("Daya       : "); Serial.println(power);
    Serial.print("Energi     : "); Serial.println(energy);
    Serial.print("Suhu       : "); Serial.println(suhu);
    Serial.print("Kelembaban : "); Serial.println(kelembaban);
    Serial.print("Gas        : "); Serial.println(gasValue);

    // Buzzer
    if (gasValue > 1000) {
      digitalWrite(BUZZER_PIN, HIGH);
      Serial.println("Peringatan! Suhu tinggi atau gas berlebih!!!");
    } else {
      digitalWrite(BUZZER_PIN, LOW);
    }

    HTTPClient https;
    https.begin(client, serverName);
    https.addHeader("Content-Type", "application/x-www-form-urlencoded");

    String postData = "tegangan=" + String(voltage) +
                      "&arus=" + String(current) +
                      "&daya=" + String(power) +
                      "&energi=" + String(energy) +
                      "&suhu=" + String(suhu) +
                      "&kelembaban=" + String(kelembaban) +
                      "&gas=" + String(gasValue);

    int httpResponseCode = https.POST(postData);

    if (httpResponseCode > 0) {
      Serial.print("Data terkirim, kode respons: ");
      Serial.println(httpResponseCode);
    } else {
      Serial.print("Gagal kirim data, error: ");
      Serial.println(httpResponseCode);
    }
    https.end();

    HTTPClient httpsRelay;
    httpsRelay.begin(client, getURL);
    int httpCode = httpsRelay.GET();

    if (httpCode == 200) {
      String payload = httpsRelay.getString();
      Serial.println("Data dari server: " + payload);

      StaticJsonDocument<256> doc;
      DeserializationError error = deserializeJson(doc, payload);

      if (!error) {
        for (int i = 1; i <= 10; i++) {
          String key = "relay" + String(i);
          int state = doc[key];
          digitalWrite(relayPins[i - 1], state == 1 ? LOW : HIGH);
        }
      } else {
        Serial.println("JSON parse error");
      }
    } else {
      Serial.println("Gagal ambil status relay, kode: " + String(httpCode));
    }

    httpsRelay.end();

  } else {
    Serial.println("WiFi terputus, mencoba reconnect...");
    WiFi.begin(ssid, password);
  }

  delay(2000);
}
